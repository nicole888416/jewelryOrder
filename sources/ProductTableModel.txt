package util;

import model.Product;
import javax.swing.table.AbstractTableModel;
import java.util.ArrayList;
import java.util.List;

public class ProductTableModel extends AbstractTableModel {
    // 欄位名稱建議直接寫中文
    private final String[] columns = {"選取", "編號", "名稱", "價格", "庫存"};
    private List<Product> data = new ArrayList<>();

    public void setData(List<Product> list) {
        this.data = (list == null) ? new ArrayList<>() : new ArrayList<>(list);
        // 通知 JTable 重新抓取資料並重新繪製
        fireTableDataChanged(); 
    }

    public Product getRow(int row) {
        if (row < 0 || row >= data.size()) return null;
        return data.get(row);
    }

    @Override public int getRowCount() { return data.size(); }
    @Override public int getColumnCount() { return columns.length; }
    @Override public String getColumnName(int column) { return columns[column]; }

    @Override
    public Class<?> getColumnClass(int columnIndex) {
        if (columnIndex == 0) return Boolean.class; // 這會讓 JTable 畫出 Checkbox
        if (columnIndex == 1 || columnIndex >= 3) return Integer.class;
        return String.class;
    }

    @Override
    public boolean isCellEditable(int rowIndex, int columnIndex) {
        return columnIndex == 0; // 只有 Checkbox 可點
    }

    @Override
    public Object getValueAt(int rowIndex, int columnIndex) {
        Product p = data.get(rowIndex);
        switch (columnIndex) {
            case 0: return p.isSelected();
            case 1: return p.getId();
            case 2: return p.getName();
            case 3: return p.getPrice();
            case 4: return p.getStock();
            default: return null;
        }
    }

    @Override
    public void setValueAt(Object aValue, int rowIndex, int columnIndex) {
        if (columnIndex == 0 && aValue instanceof Boolean) {
            data.get(rowIndex).setSelected((Boolean) aValue);
            fireTableCellUpdated(rowIndex, columnIndex);
        }
    }
}
